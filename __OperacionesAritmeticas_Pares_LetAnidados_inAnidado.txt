   fun add(x)
        fst(x)  + snd(x)
     end
     
     fun subtract(x)
         fst(x) - snd(x)
     end
     
     fun multiply(x)
        fst(x)  * snd(x)
     end
     
     fun divide(x)
        fst(x)  / snd(x)
     end
     
      fun resto(x)
        fst(x)  % snd(x)
     end
