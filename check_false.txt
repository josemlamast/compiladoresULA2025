@@ CHECKTRU - Sintaxis correcta pero tipos incorrectos
@@ Ejemplos que fallan en type checking

@@ 1. OPERACIONES ARITMÉTICAS CON TIPOS INCOMPATIBLES
5 + "hola"
2.5 - true
10 * false
8 / "texto"

@@ 2. OPERACIONES LÓGICAS CON TIPOS INCOMPATIBLES
5 and true
"hola" or false
2.5 xor true
not 5

@@ 3. COMPARACIONES CON TIPOS INCOMPATIBLES
5 > "hola"
true < 2.5
"texto" == 10
false != 3.14

@@ 4. OPERACIONES CON ARRAYS Y TIPOS INCORRECTOS
head(5)
tail("hola")
length(true)

@@ 5. OPERACIONES CON PAIRS Y TIPOS INCORRECTOS
fst(5)
snd("hola")
fst(true)

@@ 6. CONCATENACIÓN CON TIPOS INCOMPATIBLES
<+>(5, [2, 8])
<+>([5], "hola")
<+>("texto", [1, 2])

@@ 7. ELIMINACIÓN CON TIPOS INCOMPATIBLES
<->(5, 0)
<->("hola", 1)
<->(true, 2)

@@ 8. LET CON TIPOS INCOMPATIBLES
let x = 5 in x + "hola" end
let x = true in x * 2 end
let x = "texto" in x - 1 end

@@ 9. IF CON TIPOS DE RETORNO INCOMPATIBLES
if (5 > 3) 10 else "hola" end
if (true) 2.5 else false end
if (false) "texto" else 5 end

@@ 10. FUNCIONES CON TIPOS INCOMPATIBLES
fun test(x) x + "hola" end
fun test(x) x * true end
fun test(x) x / false end

@@ 11. LLAMADAS CON TIPOS INCORRECTOS
fun test(x) x + 1 end
test("hola")

fun test(x) x * 2 end
test(true)

fun test(x) head(x) end
test(5)

@@ 12. ARRAYS CON TIPOS MIXTOS INCOMPATIBLES
[5, "hola", true]
[2.5, false, "texto"]
[true, 10, 3.14]

@@ 13. PAIRS CON TIPOS INCOMPATIBLES PARA OPERACIONES
(5, "hola") + (2, 3)
(true, false) * (1, 2)
("texto", 5) / (2, 3)

@@ 14. FUNCIONES RECURSIVAS CON TIPOS INCORRECTOS
fun factorial(x) 
    if (x == 0) 
        "uno"
    else
        x * factorial(x - 1)
    end
end

@@ 15. EXPRESIONES COMPLEJAS CON TIPOS INCOMPATIBLES
let x = 5 in 
    let y = "hola" in 
        x + y
    end
end

@@ 16. COMPARACIONES DE ARRAYS CON TIPOS DIFERENTES
[5, 2, 8] > [1, 2, 3]
[true, false] == [1, 0]
["a", "b"] < [1, 2]

@@ 17. OPERACIONES CON TIPOS DE RETORNO INCONSISTENTES
fun test(x) 
    if (x > 0) 
        5
    else
        "negativo"
    end
end

@@ 18. FUNCIONES CON PARÁMETROS DE TIPO INCORRECTO
fun procesarArray(arr) 
    head(arr) + tail(arr)
end

procesarArray(5)

@@ 19. EXPRESIONES CON TIPOS MIXTOS EN OPERACIONES
5 + [2, 8]
"hola" * [1, 2]
true / [3, 4]

@@ 20. FUNCIONES CON RETORNO DE TIPO INCORRECTO
fun obtenerMaximo(arr) 
    if (head(arr) > head(tail(arr))) 
        "mayor"
    else
        "menor"
    end
end

@@ 21. LET CON TIPOS INCOMPATIBLES EN EXPRESIONES COMPLEJAS
let x = 5 in 
    if (x > 0) 
        x + "positivo"
    else
        x - "negativo"
    end
end

@@ 22. FUNCIONES CON TIPOS INCOMPATIBLES EN RECURSION
fun contar(x) 
    if (x == 0) 
        "cero"
    else
        contar(x - 1) + 1
    end
end

@@ 23. OPERACIONES CON TIPOS DE ARRAY INCORRECTOS
<+>(5, [2, 8])
<->("hola", 0)
head(5)

@@ 24. EXPRESIONES CON TIPOS INCOMPATIBLES EN COMPARACIONES
if (5 > "hola") 
    10
else
    "texto"
end

@@ 25. FUNCIONES CON TIPOS INCOMPATIBLES EN OPERACIONES ARITMÉTICAS
fun sumar(x) 
    x + "suma"
end

fun multiplicar(x) 
    x * true
end

fun dividir(x) 
    x / false
end
