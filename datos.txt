(5 + 3) * 2

# 3. Operaciones lógicas
true and false or true

# 4. Operaciones de comparación
10 > 5 and 3 < 7

# 5. Binding de variables simple
let x = 10 in x + 5

# 6. Binding de variables anidado

# 7. Expresión condicional con else
if(false) 10 
 else 20

 itos(42) # " es la respuesta"

not false

# 19. XOR lógico
true xor false

5 <= 10 and 3 >= 1
